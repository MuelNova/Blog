import { React } from "react";
import LinkCard from "@site/src/contents/pages/links/_Links.tsx";
import LinksVals from "./_linksVals.tsx";
import GiscusComment from "@site/src/theme/components/GiscusComment/GiscusComment";

export default function Links() {
  const shuffledLinks = [...LinksVals].sort(() => Math.random() - 0.5);

  return (
    <main>
      {shuffledLinks.map((link, index) => (
        <LinkCard key={index} {...link} />
      ))}
      <br />
      <GiscusComment />
    </main>
  );
}
